------------------------
ID: advanced_pricing_table
Type: SharedSlice
Description: AdvancedPricingTable

  VARIATION: Default (default)
    Primary Fields:
      - section_label (Text) — Section Label
      - title (Text) — Title
      - subtitle (StructuredText) — Subtitle
      - section_ctas (Link) (repeating) — Section CTAs
      - plans (Group) — Plans
        - name (Text) — Plan name
        - type_key (Text) — Type Key
        - description (StructuredText) — Description
        - price (Number) — Price
        - pricetext (Text) — PriceText
        - show_prices (Boolean) — Show Prices
        - price_suffix (Text) — Price Suffix (e.g. /month)
        - cta (Link) — Call to Action
      - features (Group) — Features
        - feature_name (Text) — Feature Name
        - feature_description (StructuredText) — Feature Description
        - plan_values (Group) — Plan Values
          - plan_key (Text) — Plan Type Key (must match plan.type_key)
          - value_type (Select) — Value Type
          - bool_value (Boolean) — Yes/No
          - text_value (Text) — Text Value (e.g. '10 seats', 'Unlimited')
